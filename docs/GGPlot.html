                      dv_means,
                      dv_SEs)

bar graph with error bar

ggplot(plot_df, aes(x=factor_one,y=dv_means))+
  geom_bar(stat="identity")+
  geom_errorbar(aes(ymin=dv_means-dv_SEs, 
                    ymax=dv_means+dv_SEs),
